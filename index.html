function pickPlayer(name, type, team, abbr, key) {

  const who = whoseTurn();

  stopPickTimer();

  

  G.picks[who].push({name,type,team,abbr});

  if(type==='bat') G.batPicked[who]++;

  else G.bowlPicked[who]++;

  

  // Update the row

  const row = document.getElementById('row-'+key);

  if(row) {

    row.className = `player-row owner-${who} just-picked`;

    const oBadge = who==='a' ? `<span class="owner-badge a">${G.playerA}</span>` : `<span class="owner-badge b">${G.playerB}</span>`;

    row.innerHTML = `<div style="display:flex;align-items:center;flex:1;">

      <div class="pr-avatar ${who}">${abbr}</div>

      <div><div class="pr-name strikethrough">${name}</div><div class="pr-meta">${type==='bat'?'Batsman':'Bowler'} · ${team}</div></div>

    </div>

    <div class="pr-right">${oBadge}</div>`;

  }

  

  G.currentPick++;

  

  // Check phase transition

  if(G.batPicked.a+G.batPicked.b===6 && G.phase==='bat') G.phase='bowl';

  

  // Check done

  if(G.currentPick>=G.totalPicks) {

    G.draftDone = true;

    document.getElementById('turn-banner').className='turn-banner their-turn';

    document.getElementById('turn-title').textContent='Draft Complete!';

function simulateOpponentPick() {

  if(G.draftDone || whoseTurn()!=='b') return;

  const pool = G.phase==='bat' ? PLAYERS.bat : PLAYERS.bowl;

  const available = pool.filter(p => !G.picks.a.find(x=>x.name===p.name) && !G.picks.b.find(x=>x.name===p.name));

  if(!available.length) return;

  

  const p = available[Math.floor(Math.random()*available.length)];

function calcScore(picks) {

  let total=0;const lines=[];

  picks.forEach(p=>{

    const s = LIVE_SCORES[p.name]||{runs:0,wickets:0,catches:0};

    let pt=0;

    if(p.type==='bat'){pt=(s.runs||0)*10+(s.catches||0)*50;lines.push({name:p.name,det:`${s.runs||0} runs${s.catches?', '+s.catches+' catch':''}`,total:pt});}

    else{pt=(s.wickets||0)*200+(s.catches||0)*50;lines.push({name:p.name,det:`${s.wickets||0} wkts${s.catches?', '+s.catches+' catch':''}`,total:pt});}

    total+=pt;

  });

  return{total,lines};

}
